import heapq
from collections import defaultdict, Counter

# Define the node structure for the Huffman tree
class HuffmanNode:
    def __init__(self, char=None, freq=0):
        self.char = char
        self.freq = freq
        self.left = None
        self.right = None

    # For priority queue to compare nodes
    def __lt__(self, other):
        return self.freq < other.freq

# Build the Huffman Tree
def build_huffman_tree(text):
    freq = Counter(text)
    priority_queue = [HuffmanNode(char, fr) for char, fr in freq.items()]
    heapq.heapify(priority_queue)

    while len(priority_queue) > 1:
        node1 = heapq.heappop(priority_queue)
        node2 = heapq.heappop(priority_queue)
        merged = HuffmanNode(None, node1.freq + node2.freq)
        merged.left = node1
        merged.right = node2
        heapq.heappush(priority_queue, merged)

    return priority_queue[0]

# Generate the codes from the tree
def generate_codes(root):
    codes = {}

    def dfs(node, code=""):
        if node:
            if node.char is not None:
                codes[node.char] = code
            dfs(node.left, code + "0")
            dfs(node.right, code + "1")

    dfs(root)
    return codes

# Huffman Encoding
def huffman_encoding(text):
    root = build_huffman_tree(text)
    codes = generate_codes(root)
    encoded_text = ''.join(codes[char] for char in text)
    return encoded_text, codes, root

# Huffman Decoding
def huffman_decoding(encoded_text, root):
    decoded = []
    node = root
    for bit in encoded_text:
        node = node.left if bit == '0' else node.right
        if node.char is not None:
            decoded.append(node.char)
            node = root
    return ''.join(decoded)

# Example Usage
if __name__ == "__main__":
    text = "huffman coding is greedy"
    encoded, code_map, tree = huffman_encoding(text)
    print("Huffman Codes:", code_map)
    print("Encoded String:", encoded)
    print("Decoded String:", huffman_decoding(encoded, tree))

# 🔹 1. Concept: Huffman Coding

👉 Huffman Coding is a **lossless data compression technique**

### Goal:

* Reduce size of data
* Assign **shorter codes to frequent characters**
* Assign **longer codes to rare characters**

✔ No data loss
✔ Efficient encoding

---

# 🔹 2. Key Idea (Greedy Strategy)

👉 Always combine **two lowest frequency nodes first**

✔ This ensures:

* Minimum total encoding cost
* Optimal prefix-free codes

---

# 🔹 3. Important Properties

✔ **Prefix-free codes** → No code is prefix of another
✔ Based on **binary tree (Huffman Tree)**
✔ Uses **priority queue (min heap)**
