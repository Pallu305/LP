from queue import PriorityQueue

class Node:
    def __init__(self, level, profit, weight, bound):
        self.level = level
        self.profit = profit
        self.weight = weight
        self.bound = bound

    def __lt__(self, other):
        return self.bound > other.bound  # Higher bound gets priority


def bound(node, n, capacity, weights, values):
    if node.weight >= capacity:
        return 0

    profit_bound = node.profit
    j = node.level + 1
    total_weight = node.weight

    # Add items while there's capacity
    while j < n and total_weight + weights[j] <= capacity:
        total_weight += weights[j]
        profit_bound += values[j]
        j += 1

    # Add fraction of the next item
    if j < n:
        profit_bound += (capacity - total_weight) * (values[j] / weights[j])

    return profit_bound


def knapsack_branch_and_bound(weights, values, capacity):
    n = len(values)

    # Combine and sort items by value/weight ratio
    items = list(zip(values, weights))
    items.sort(key=lambda x: x[0] / x[1], reverse=True)

    values = [item[0] for item in items]
    weights = [item[1] for item in items]

    q = PriorityQueue()

    root = Node(-1, 0, 0, 0)
    root.bound = bound(root, n, capacity, weights, values)
    q.put(root)

    max_profit = 0

    while not q.empty():
        node = q.get()

        if node.bound > max_profit:
            level = node.level + 1

            if level < n:
                # Include the current item
                next_node = Node(
                    level,
                    node.profit + values[level],
                    node.weight + weights[level],
                    0
                )

                if next_node.weight <= capacity:
                    max_profit = max(max_profit, next_node.profit)

                next_node.bound = bound(next_node, n, capacity, weights, values)

                if next_node.bound > max_profit:
                    q.put(next_node)

                # Exclude the current item
                next_node = Node(level, node.profit, node.weight, 0)
                next_node.bound = bound(next_node, n, capacity, weights, values)

                if next_node.bound > max_profit:
                    q.put(next_node)

    return max_profit


if __name__ == "__main__":

    weights = [10, 20, 30]
    capacity = 50

    max_value = knapsack_branch_and_bound(weights, values, capacity)

    print("Maximum value in the knapsack (Branch and Bound):", max_value)

# 🔹 1. Concept: Branch and Bound

👉 It is an **optimization technique** used to solve problems like 0/1 Knapsack efficiently.

### Idea:

* Represent solution as a **state space tree**
* Each node = decision (include/exclude item)
* Use **bound (upper estimate)** to decide:

  * Explore node ✅
  * Prune node ❌

---

# 🔹 2. Why Use It?

* Brute force → (O(2^n)) ❌
* DP → good but uses more memory
* Branch & Bound → **faster in practice due to pruning**
