import heapq

def tsp_branch_and_bound(cost_matrix):
    n = len(cost_matrix)

    # --- Matrix Reduction Function ---
    def reduce_matrix(matrix):
        mat = [row[:] for row in matrix]
        reduction_cost = 0

        # Row reduction
        for i in range(n):
            row_min = min(mat[i])
            if row_min != float('inf') and row_min > 0:
                reduction_cost += row_min
                for j in range(n):
                    if mat[i][j] != float('inf'):
                        mat[i][j] -= row_min

        # Column reduction
        for j in range(n):
            col = [mat[i][j] for i in range(n)]
            col_min = min(col)
            if col_min != float('inf') and col_min > 0:
                reduction_cost += col_min
                for i in range(n):
                    if mat[i][j] != float('inf'):
                        mat[i][j] -= col_min

        return mat, reduction_cost

    # Initial reduction
    reduced_matrix, cost = reduce_matrix(cost_matrix)

    # Priority Queue: (cost, level, current_node, path, matrix)
    pq = []
    heapq.heappush(pq, (cost, 0, 0, [0], reduced_matrix))

    min_cost = float('inf')
    best_path = []

    while pq:
        cost, level, current, path, matrix = heapq.heappop(pq)

        # Prune
        if cost >= min_cost:
            continue

        # If all cities visited
        if level == n - 1:
            final_cost = cost + cost_matrix[current][0]
            if final_cost < min_cost:
                min_cost = final_cost
                best_path = path + [0]
            continue

        # Explore next cities
        for next_city in range(n):
            if next_city not in path and matrix[current][next_city] != float('inf'):

                # Copy matrix
                new_matrix = [row[:] for row in matrix]

                # Set row and column to infinity
                for k in range(n):
                    new_matrix[current][k] = float('inf')
                    new_matrix[k][next_city] = float('inf')

                # Prevent early return to start
                new_matrix[next_city][0] = float('inf')

                # Reduce new matrix
                reduced_new_matrix, reduction_cost = reduce_matrix(new_matrix)

                new_cost = cost + matrix[current][next_city] + reduction_cost

                if new_cost < min_cost:
                    heapq.heappush(
                        pq,
                        (new_cost, level + 1, next_city, path + [next_city], reduced_new_matrix)
                    )

    return best_path, min_cost


# --- Main Function ---
if __name__ == "__main__":
    inf = float('inf')

    cost_matrix = [
        [inf, 10, 15, 20],
        [10, inf, 35, 25],
        [15, 35, inf, 30],
        [20, 25, 30, inf]
    ]

    path, cost = tsp_branch_and_bound(cost_matrix)

    print("Minimum Cost:", cost)
    print("Best Path:", path)

# ✅ Travelling Salesman Problem (TSP) using Branch and Bound

---

# 🔷 1. Problem Definition

The **Travelling Salesman Problem (TSP)** is:

> Given a set of cities and distances between them, find the **shortest possible route** that:

* Visits each city exactly once
* Returns to the starting city

---

# 🔷 2. Mathematical Representation

* Represented using a **cost matrix**
* If there are `n` cities → matrix size = `n × n`

[
C[i][j] = \text{cost from city i to city j}
]

* Diagonal values = ∞ (no self-loop)

---

# 🔷 3. Types of TSP

| Type       | Description        |
| ---------- | ------------------ |
| Symmetric  | Distance i→j = j→i |
| Asymmetric | Distance differs   |

👉 Your code solves **symmetric TSP**

---

# 🔷 4. Brute Force Approach

* Try all permutations
* Time complexity:

[
O(n!)
]

❌ Not efficient for large `n`

---

# 🔷 5. Branch and Bound Approach

### ✔ Idea:

* Reduce search space
* Avoid unnecessary paths

---

## 🔹 Components

### 1. Branching

* Expanding nodes (trying next cities)

### 2. Bounding

* Calculate **lower bound cost**
* If bound > best solution → prune

### 3. Pruning

* Stop exploring bad paths early

---

# 🔷 6. Lower Bound (Core Concept)

👉 Lower bound = **minimum possible cost from current state**

Used to:

* Decide whether to explore a path

---

# 🔷 7. Matrix Reduction

### Purpose:

* Convert matrix → easier to estimate cost

---

## 🔹 Row Reduction

[
\text{Subtract minimum value from each row}
]

## 🔹 Column Reduction

[
\text{Subtract minimum value from each column}
]

---

👉 Reduction cost = sum of all subtracted values
👉 This becomes the **initial lower bound**

---

# 🔷 8. State Space Tree

* Each node = partial path
* Root = starting city
* Leaves = complete tours

---

# 🔷 9. Algorithm Steps

### Step 1: Input cost matrix

### Step 2: Perform matrix reduction

### Step 3: Initialize priority queue

### Step 4: Repeat until queue empty:

* Select node with minimum cost
* If complete path → update solution
* Else:

  * Generate children (branching)
  * Reduce matrix
  * Compute new bound
  * Add to queue if promising

---

# 🔷 10. Priority Queue Role

* Stores nodes as:
  [
  (\text{cost}, \text{level}, \text{current node}, \text{path})
  ]

👉 Always expands **minimum cost node first**

---

# 🔷 11. Cost Calculation

[
\text{Total Cost} = \text{current cost} + \text{edge cost} + \text{reduction cost}
]

---

# 🔷 12. Time Complexity

* Worst case:
  [
  O(n!)
  ]

* But practically:
  👉 Much faster due to pruning

---

# 🔷 13. Space Complexity

[
O(n^2)
]

* For storing matrices and nodes

---

# 🔷 14. Advantages

✔ Faster than brute force
✔ Guarantees optimal solution
✔ Reduces unnecessary computations

# 🔷 15. Disadvantages
